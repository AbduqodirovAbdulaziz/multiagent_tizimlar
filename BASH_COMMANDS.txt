cd ~

# 2. REPOSITORY CLONE (agar mavjud bo'lsa, o'tkazib yuboring)

# 3. LOYIHA PAPKASIGA O'TISH
cd multiagent_tizimlar

# 4. VIRTUAL ENVIRONMENT YARATISH
mkvirtualenv --python=/usr/bin/python3.10 multiagent

# 5. VIRTUAL ENVIRONMENT FAOLLASHTIRISH
workon multiagent

# 6. PIP YANGILASH
pip install --upgrade pip

# 7. DEPENDENCIES O'RNATISH
pip install -r requirements/base.txt

# 8. LOGS PAPKASINI YARATISH
mkdir -p logs
touch logs/medical_mas.log

# 9. .ENV FAYL YARATISH
cat > .env << 'EOF'
DJANGO_SETTINGS_MODULE=config.settings.pythonanywhere
SECRET_KEY=django-insecure-change-this-in-production-12345
DEBUG=False
PYTHONANYWHERE_USERNAME=multiagent
DJANGO_ALLOWED_HOSTS=multiagent.pythonanywhere.com,localhost
EOF

# 10. DATABASE MIGRATE

import os
import sys

username = 'multiagent'
project_home = f'/home/{username}/multiagent_tizimlar'

if project_home not in sys.path:
    sys.path.insert(0, project_home)

os.environ['DJANGO_SETTINGS_MODULE'] = 'config.settings.pythonanywhere'

env_file = os.path.join(project_home, '.env')
if os.path.exists(env_file):
    with open(env_file) as f:
        for line in f:
            line = line.strip()
            if line and not line.startswith('#') and '=' in line:
                key, _, value = line.partition('=')
                os.environ.setdefault(key.strip(), value.strip())

from django.core.wsgi import get_wsgi_application
application = get_wsgi_application()
